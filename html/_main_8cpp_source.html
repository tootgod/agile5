105 void AMain::Tick(float DeltaTime)
106 {
107  Super::Tick(DeltaTime);
108 
109  if (MovementStatus == EMovementStatus::EMS_Dead) return;
110 
111  float DeltaStamina = StaminaDrainRate * DeltaTime;
112 
113  switch (StaminaStatus)
114  {
115  case EStaminaStatus::ESS_Normal:
116  if (bShiftKeyDown)
117  {
118  if (Stamina - DeltaStamina <= MinSprintStamina)
119  {
120  SetStaminaStatus(EStaminaStatus::ESS_BelowMinimum);
121  Stamina -= DeltaStamina;
122  }
123  else
124  {
125  Stamina -= DeltaStamina;
126  }
127  if (bMovingForward || bMovingRight)
128  {
129  SetMovementStatus(EMovementStatus::EMS_Sprinting);
130  }
131  else
132  {
133  SetMovementStatus(EMovementStatus::EMS_Normal);
134  }
135  }
136  else // Shift key up
137  {
138  if (Stamina + DeltaStamina >= MaxStamina)
139  {
140  Stamina = MaxStamina;
141  }
142  else
143  {
144  Stamina += DeltaStamina;
145  }
146  SetMovementStatus(EMovementStatus::EMS_Normal);
147  }
148  break;
149  case EStaminaStatus::ESS_BelowMinimum:
150  if (bShiftKeyDown)
151  {
152  if (Stamina - DeltaStamina <= 0.f)
153  {
154  SetStaminaStatus(EStaminaStatus::ESS_Exhausted);
155  Stamina = 0;
156  SetMovementStatus(EMovementStatus::EMS_Normal);
157  }
158  else
159  {
160  Stamina -= DeltaStamina;
161  if (bMovingForward || bMovingRight)
162  {
163  SetMovementStatus(EMovementStatus::EMS_Sprinting);
164  }
165  else
166  {
167  SetMovementStatus(EMovementStatus::EMS_Normal);
168  }
169  }
170  }
171  else // Shift key up
172  {
173  if (Stamina + DeltaStamina >= MinSprintStamina)
174  {
175  SetStaminaStatus(EStaminaStatus::ESS_Normal);
176  Stamina += DeltaStamina;
177  }
178  else
179  {
180  Stamina += DeltaStamina;
181  }
182  SetMovementStatus(EMovementStatus::EMS_Normal);
183  }
184  break;
185  case EStaminaStatus::ESS_Exhausted:
186  if (bShiftKeyDown)
187  {
188  Stamina = 0.f;
189  }
190  else // Shift key up
191  {
192  SetStaminaStatus(EStaminaStatus::ESS_ExhaustedRecovering);
193  Stamina += DeltaStamina;
194  }
195  SetMovementStatus(EMovementStatus::EMS_Normal);
196  break;
197  case EStaminaStatus::ESS_ExhaustedRecovering:
198  if (Stamina + DeltaStamina >= MinSprintStamina)
199  {
200  SetStaminaStatus(EStaminaStatus::ESS_Normal);
201  Stamina += DeltaStamina;
202  }
203  else
204  {
205  Stamina += DeltaStamina;
206  }
207  SetMovementStatus(EMovementStatus::EMS_Normal);
208  break;
209  default:
210  ;
211  }
212 
213  if (bInterpToEnemy && CombatTarget)
214  {
215  FRotator LookAtYaw = GetLookAtRotationYaw(CombatTarget->GetActorLocation());
216  FRotator InterpRotation = FMath::RInterpTo(GetActorRotation(), LookAtYaw, DeltaTime, InterpSpeed);
217 
218  SetActorRotation(InterpRotation);

223  CombatTargetLocation = CombatTarget->GetActorLocation();
